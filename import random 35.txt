import random

# Simple encryption

def encrypt(plaintext, key):
ciphertext = ""
for i in range(len(plaintext)):
if plaintext[i].isalpha():
shift = key[i]
c = chr((ord(plaintext[i].upper()) - 65 + shift) % 26 + 65)
ciphertext += c
else:
ciphertext += plaintext[i]
return ciphertext

# Simple decryption

def decrypt(ciphertext, key):
plaintext = ""
for i in range(len(ciphertext)):
if ciphertext[i].isalpha():
shift = key[i]
p = chr((ord(ciphertext[i].upper()) - 65 - shift) % 26 + 65)
plaintext += p
else:
plaintext += ciphertext[i]
return plaintext

# Example usage

plaintext = "HELLO"
key = [random.randint(0,26) for _ in range(len(plaintext))]

print("Plaintext:", plaintext)
print("Key:", key)
ciphertext = encrypt(plaintext, key)
print("Ciphertext:", ciphertext)
print("Decrypted:", decrypt(ciphertext, key))

output

Key: [3, 19, 5, 12, 0]
